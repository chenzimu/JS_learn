<script>
        //求和
        Array.prototype.sum = function () {
            var sum = 0;
            for(var i = 0;i < this.length; i++){
                sum += parseInt(this[i]);
            }
            return sum;
        }
        //求最大值
        Array.prototype.maxValue = function(){
            var maxValue = Number.MIN_VALUE;
            for(var i = 0;i < this.length;i++){
                if(parseInt(this[i]) > maxValue){
                    maxValue = this[i];
                }
            }
            return maxValue;
        }

        var arr = [12,3,4,6,2,32];
        console.log(arr.sum());
        console.log(arr.maxValue());

    </script>
